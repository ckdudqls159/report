3. 펜 만들기 스타일 크기 컬러

 

void CPen7Dlg::OnMouseMove(UINT nFlags, CPoint point)

{

	if (nFlags == MK_LBUTTON) {

		CClientDC dc(this);

		CPen pen(PS_SOLID, 16, RGB(0, 255, 0));

		dc.SelectObject(&pen);

		dc.MoveTo(pnt);

		dc.LineTo(point);

	}

	pnt = point;

	CDialogEx::OnMouseMove(nFlags, point);

}
